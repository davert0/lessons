// Хрупкая jwsSignature,неотраженная в сигнатуре, изменение body между подписанием и request.Post сломает вызов

func (a *Adapter) callPost(
	ctx context.Context,
	callData *callData,
	partnerID string,
	accessToken []byte,
	reqBody []byte,
) ([]byte, error) {
	jwsSignature, err := a.jwtProvider.BuildDetachedJWS(ctx, partnerID, reqBody)
	if err != nil {
		log.Err(ctx, "build jws error", log.Error(err))

		return nil, models.ErrServiceUnexpected
	}

	interactionUUID, err := a.gen.UUIDv4()
	if err != nil {
		log.Err(ctx, "gen uuid error", log.Error(err))

		return nil, models.ErrServiceUnexpected
	}

	interactionID := interactionUUID.String()

	ctx = log.WithFields(ctx, log.String(account.HeaderFapiInteractionID, interactionID))

	client, err := a.getClient(ctx, partnerID)
	if err != nil {
		return nil, err
	}

	httpResponse, err := client.Post(ctx, callData.path,
		rest.WithServiceMethod(callData.serviceMethod),
		rest.WithRequestBody(reqBody),
		rest.WithRequestHeader(rest.HeaderAuthorization, rest.MakeBearer(slice.ToString(accessToken))),
		rest.WithRequestHeader(account.HeaderJwsSignature, slice.ToString(jwsSignature)),
		rest.WithRequestHeader(account.HeaderFapiInteractionID, interactionID),
		rest.WithRequestHeader(rest.HeaderContentType, rest.JSONContentType),
		rest.WithRequestHeader(rest.HeaderAccept, rest.JSONContentType),
		rest.WithRequestHeader(clientIDHeader, partnerID),
	)
	if err != nil {
		log.Err(ctx, "call post error", log.Error(err))

		return nil, models.ErrServiceUnavailable
	}

	defer rest.CloseResponse(ctx, httpResponse)
}
