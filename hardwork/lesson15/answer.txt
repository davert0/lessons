        return &Adapter{
            client:         redisCluster,
            expirationTime: expirationTime,
            cache:          cacheRedis,
            scripts:        make(map[string]*redis.Script), // ← здесь
        }
    }

    func (a *Adapter) script(src string) *redis.Script {
        a.scriptMu.Lock()
        defer a.scriptMu.Unlock()
        s, ok := a.scripts[src]   // больше нет проверки на nil
        if !ok { s = redis.NewScript(src); a.scripts[src] = s }
        return s
    }

После: генерик функция для пагинации данных

  func fetchAllPages[T any](
      ctx context.Context,
      fetch func(ctx context.Context, page int) (data []T, totalPages int, err error),
  ) ([]T, error) {
      var all []T
      for page := 1; ; page++ {
          data, total, err := fetch(ctx, page)
          if err != nil { return nil, err }
          all = append(all, data...)
          if page >= total { return all, nil }   // единственное условие выхода
      }
  }
