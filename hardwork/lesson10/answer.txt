// 4. Функция проверяет userInfo, принимает бизнес решение, отзывает согласие

// До
	if a.revokeConsentIfUserInfoFail(ctx, writer, consentID, partnerID, requestBody.SiebelID,
		idTok.Subject, clientID, tokenManager) {
		return
	}

// После
	result := a.checkUserInfo(ctx, consentID, partnerID, siebelID, idTokenSub, clientID, tm)
    if result.Err != nil {
        log.Err(ctx, "user info error", log.Error(result.Err))
    }

    needsRevoke := (!result.NameMatch || !result.PhoneMatch || result.Err != nil) &&
        a.features.RevokeConsentIfUserInfoFail

    if needsRevoke {
        if err := a.revokeConsent(ctx, consentID, partnerID); err != nil {
            log.Err(ctx, "Delete consent error", log.Error(err),
                log.String("consentID", consentID), log.String("partnerID", partnerID))
        }

        failure := classifyUserInfoFailure(result)
        switch failure {
        case UserInfoInternalError:
            a.sendErrInternalServerError(ctx, w)
        case UserInfoNoMatchNamePhone:
            a.sendErrUserNoMatchNamePhone(ctx, w)
        case UserInfoNoMatchName:
            a.sendErrUserNoMatchName(ctx, w)
        case UserInfoNoMatchPhone:
            a.sendErrUserNoMatchPhone(ctx, w)
        }

        if failure != UserInfoInternalError {
            if err := a.consentsProvider.MarkUserInfoFailed(ctx, consentID); err != nil {
                log.Err(ctx, "could not mark consent as user info failed", log.Error(err))
            }
        }
        return
    }
